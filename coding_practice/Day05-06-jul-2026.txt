## 🔹 Jenkinsfile — With Git Integration and Testing

```groovy

    agent any

        stage('Clone Repository') {

            }
        }

        }

            }
        }
        stage('deploy') {
            steps {
                echo 'Deploying application'
            }
        }
     }
}

🔹 Ansible — Create User and Set Permissions
- name: Create User and Set permission
  host: all
  become: yes
  tasks:
    - name: Create user
      user:
        name: pranay
        stage: present
        shell: /bin/bash
    - name: Set permission for the user
      file:
        path: /home/pranay
        owner: pranay
        group: sudo
        mode: '0755'
