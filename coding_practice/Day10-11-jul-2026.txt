## 🔹 Ansible — Create Docker Containers for Multiple Applications

```yaml
- name: Setup Docker Containers for Multiple Apps
  hosts: all
  become: yes
  tasks:
    - name: Pull Web App Image
      docker_image:
        name: pranay-web-app
        source: pull

    - name: Pull Database App Image
      docker_image:
        name: pranay-database-app
        source: pull

    - name: Run Web App Container
      docker_container:
        name: web-app
        image: pranay-web-app
        state: started
        published_ports:
          - "8080:80"

    - name: Run Database App Container
      docker_container:
        name: database-app
        image: pranay-database-app
        state: started
        published_ports:
          - "3306:3306"

🔹 Kubernetes — Deploying StatefulSet
apiVersion: app/v1
kind: StatefulSet
metadata:
  name: pranay-db
spec:
  serviceName: "pranay-db"
  replicas: 3
  selector:
    matchLabels:
      app: pranay-db
  template:
    metadata:
      labels:
        app: pranay-db
    spec:
      containers:
        - name: db
          image: mysql:8.0
          volumeMounts:
            - mountPath: /var/lib/mysql
              name: mysql-storage
   volumeClaimTemplates:
     - metadata:
         name: mysql-storage
       spec:
         accessModes: ["ReadWriteOnce"]
         resources:
           requests:
             storage: 1Gi

🔹 Jenkinsfile — Build and Deploy to Kubernetes
pipeline {
    agent any
    environment {
        IMAGE_NAME = 'pranay-web-app'
    }
    stages {
        stage('Build Docker Images') {
            steps {
                script {
                    docker.build("${env.IMAGE_NAME}")
                }
            }
        }
        stage('Push Docker Image') {
            steps {
                script {
                    docker.withRegistry('https://docker.io', 'docker-credentials') {
                        docker.image("${env.IMAGE_NAME}").push('latest')
                    }
                }
            }
        }
        stage('Deploy to Kubernetes') {
            steps {
                script {
                    sh 'kubectl apply -f deployment.yaml'
                }
            }
        }
    }
}

🔹 GitLab CI/CD — Build, Push, and Deploy
stages:
  - build
  - push
  - deploy

build:
  stage: build
  script:
    - docker build -t pranay-web-app .

push:
  stage: push
  script:
    - dockerlogin -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"
    - docker push pranay-web-app

deploy:
  stage: deploy
  script:
    - kubectl apply -f kubernetes/deployment.yaml
